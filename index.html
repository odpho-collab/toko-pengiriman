🔗 Hubungkan ke Google Sheets

Data akan otomatis tersinkron ke Google Sheets Anda secara real-time dan bisa diakses dari perangkat mana pun.

1Buat Google Sheet baruBuka sheets.google.com → buat spreadsheet baru → beri nama bebas.
2Buat Apps ScriptDi Google Sheets: klik Extensions → Apps Script → hapus semua kode → paste kode berikut:
var FOLDER_ID = 'FOLDER_ID_ANDA'; // ganti ini function doPost(e) { var data = JSON.parse(e.postData.contents); var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheets()[0]; if (data.action === 'add') { var r = data.record; var fc = data.fotoCount || {}; if (sheet.getLastRow() === 0) { sheet.appendRow(['Tanggal & Jam','No. Surat Jalan','Nama Pengirim', 'Jenis Kendaraan','No. Polisi','Asal Toko', 'Link Foto Pickup','Link Foto SJ','Link Foto Supir']); } sheet.appendRow([r.waktu, r.sj, r.nama, r.jenis, r.nopol, r.toko, (fc.pickup||0)+' foto (menunggu...)', (fc.sj||0)+' foto (menunggu...)', (fc.supir||0)+' foto (menunggu...)']); } else if (data.action === 'foto') { var folder = DriveApp.getFolderById(FOLDER_ID); var subFolders = folder.getFoldersByName(data.sjNo); var sub = subFolders.hasNext() ? subFolders.next() : folder.createFolder(data.sjNo); var b64 = data.data.split(',')[1]; var blob = Utilities.newBlob( Utilities.base64Decode(b64), 'image/jpeg', data.type + '-' + (data.index+1) + '.jpg'); var file = sub.createFile(blob); file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); // Update link di sheet var lastRow = sheet.getLastRow(); for (var i = lastRow; i >= 2; i--) { if (sheet.getRange(i, 2).getValue() == data.sjNo) { var col = data.type === 'pickup' ? 7 : (data.type === 'sj' ? 8 : 9); var existing = sheet.getRange(i, col).getValue(); var newVal = (existing.indexOf('menunggu') > -1) ? file.getUrl() : existing + ' ' + file.getUrl(); sheet.getRange(i, col).setValue(newVal); break; } } } return ContentService.createTextOutput(JSON.stringify({status:'ok'})).setMimeType(ContentService.MimeType.JSON); } function doGet(e) { return ContentService.createTextOutput(JSON.stringify({status:'ok'})).setMimeType(ContentService.MimeType.JSON); }
3Deploy sebagai Web AppKlik Deploy → New Deployment → Web App → "Who has access": Anyone → Deploy → copy URL.
4Paste URL di bawah iniTempel URL dari langkah 3, lalu klik Simpan & Test.
